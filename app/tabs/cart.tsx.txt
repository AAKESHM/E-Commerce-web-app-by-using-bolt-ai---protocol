import { useEffect, useState } from 'react';
import {
  StyleSheet,
  View,
  Text,
  ScrollView,
  TouchableOpacity,
  Image,
  ActivityIndicator,
  Alert,
} from 'react-native';
import { router } from 'expo-router';
import { useCart } from '@/lib/cart';
import { theme } from '@/lib/theme';
import { formatPrice } from '@/lib/format';
import { Minus, Plus, Trash2, ShoppingBag, ArrowRight, Tag } from 'lucide-react-native';

const DELIVERY_FEE = 25;
const FREE_DELIVERY_THRESHOLD = 199;

export default function CartScreen() {
  const { items, loading, subtotal, setQuantity, removeItem } = useCart();
  const deliveryFee = subtotal >= FREE_DELIVERY_THRESHOLD || subtotal === 0 ? 0 : DELIVERY_FEE;
  const total = subtotal + deliveryFee;

  if (loading) {
    return (
      <View style={[styles.center, { backgroundColor: theme.colors.background }]}>
        <ActivityIndicator size="large" color={theme.colors.primary} />
      </View>
    );
  }

  if (items.length === 0) {
    return (
      <View style={styles.center}>
        <View style={styles.emptyIcon}>
          <ShoppingBag size={48} color={theme.colors.neutral[400]} strokeWidth={1.5} />
        </View>
        <Text style={styles.emptyTitle}>Your cart is empty</Text>
        <Text style={styles.emptySub}>Add some fresh items to get started</Text>
        <TouchableOpacity style={styles.shopBtn} onPress={() => router.push('/(tabs)/search')}>
          <Text style={styles.shopBtnText}>Browse products</Text>
        </TouchableOpacity>
      </View>
    );
  }

  return (
    <View style={styles.container}>
      <View style={styles.header}>
        <Text style={styles.headerTitle}>My Cart</Text>
        <Text style={styles.headerCount}>{items.length} items</Text>
      </View>

      {subtotal < FREE_DELIVERY_THRESHOLD && (
        <View style={styles.freeShipBanner}>
          <Tag size={14} color={theme.colors.primaryDark} />
          <Text style={styles.freeShipText}>
            Add {formatPrice(FREE_DELIVERY_THRESHOLD - subtotal)} more for FREE delivery
          </Text>
        </View>
      )}

      <ScrollView contentContainerStyle={styles.list} showsVerticalScrollIndicator={false}>
        {items.map((item) => {
          const p = item.product;
          if (!p) return null;
          return (
            <View key={item.id} style={styles.row}>
              <Image source={{ uri: p.image_url ?? '' }} style={styles.thumb} />
              <View style={styles.rowInfo}>
                <Text style={styles.rowName} numberOfLines={2}>{p.name}</Text>
                <Text style={styles.rowUnit}>per {p.unit}</Text>
                <Text style={styles.rowPrice}>{formatPrice(p.price)}</Text>
                <View style={styles.qtyRow}>
                  <TouchableOpacity
                    style={styles.qtyBtn}
                    onPress={() => setQuantity(p.id, item.quantity - 1)}
                  >
                    <Minus size={14} color={theme.colors.text} />
                  </TouchableOpacity>
                  <Text style={styles.qty}>{item.quantity}</Text>
                  <TouchableOpacity
                    style={styles.qtyBtn}
                    onPress={() => setQuantity(p.id, item.quantity + 1)}
                  >
                    <Plus size={14} color={theme.colors.text} />
                  </TouchableOpacity>
                </View>
              </View>
              <View style={styles.rowRight}>
                <TouchableOpacity onPress={() => removeItem(p.id)} hitSlop={{ top: 10, bottom: 10, left: 10, right: 10 }}>
                  <Trash2 size={18} color={theme.colors.error} />
                </TouchableOpacity>
                <Text style={styles.rowTotal}>{formatPrice(Number(p.price) * item.quantity)}</Text>
              </View>
            </View>
          );
        })}
      </ScrollView>

      <View style={styles.summary}>
        <View style={styles.summaryRow}>
          <Text style={styles.summaryLabel}>Subtotal</Text>
          <Text style={styles.summaryValue}>{formatPrice(subtotal)}</Text>
        </View>
        <View style={styles.summaryRow}>
          <Text style={styles.summaryLabel}>Delivery fee</Text>
          <Text style={styles.summaryValue}>{deliveryFee === 0 ? 'FREE' : formatPrice(deliveryFee)}</Text>
        </View>
        <View style={styles.summaryDivider} />
        <View style={styles.summaryRow}>
          <Text style={styles.summaryTotalLabel}>Total</Text>
          <Text style={styles.summaryTotalValue}>{formatPrice(total)}</Text>
        </View>
        <TouchableOpacity style={styles.checkoutBtn} onPress={() => router.push('/checkout')}>
          <Text style={styles.checkoutBtnText}>Proceed to Checkout</Text>
          <ArrowRight size={18} color="#fff" />
        </TouchableOpacity>
      </View>
    </View>
  );
}
