import { StyleSheet, View, Text, ScrollView, TouchableOpacity, Alert } from 'react-native';
import { router } from 'expo-router';
import { useAuth } from '@/lib/auth';
import { useWishlist } from '@/lib/wishlist';
import { theme } from '@/lib/theme';
import {
  Heart,
  Package,
  MapPin,
  CreditCard,
  Bell,
  HelpCircle,
  LogOut,
  ChevronRight,
  Shield,
  Star,
  User as UserIcon,
} from 'lucide-react-native';

export default function ProfileScreen() {
  const { user, signOut } = useAuth();
  const { items } = useWishlist();

  const handleSignOut = () => {
    Alert.alert('Sign out', 'Are you sure you want to sign out?', [
      { text: 'Cancel', style: 'cancel' },
      { text: 'Sign out', style: 'destructive', onPress: () => signOut() },
    ]);
  };

  const email = user?.email ?? '';
  const initials = email.slice(0, 2).toUpperCase();

  return (
    <ScrollView style={styles.container} contentContainerStyle={{ paddingBottom: 32 }} showsVerticalScrollIndicator={false}>
      <View style={styles.header}>
        <View style={styles.avatar}>
          <Text style={styles.avatarText}>{initials}</Text>
        </View>
        <Text style={styles.userName}>{email}</Text>
        <Text style={styles.userSub}>Member since 2026</Text>
      </View>

      <View style={styles.statsRow}>
        <Stat label="Orders" value="—" icon={<Package size={18} color={theme.colors.primary} />} />
        <Stat label="Wishlist" value={`${items.length}`} icon={<Heart size={18} color={theme.colors.secondary} />} />
        <Stat label="Points" value="0" icon={<Star size={18} color={theme.colors.accent} />} />
      </View>

      <Section title="Account">
        <Row icon={<UserIcon size={18} color={theme.colors.textMuted} />} label="Personal details" onPress={() => {}} />
        <Row icon={<MapPin size={18} color={theme.colors.textMuted} />} label="Delivery addresses" onPress={() => {}} />
        <Row icon={<CreditCard size={18} color={theme.colors.textMuted} />} label="Payment methods" onPress={() => {}} />
        <Row icon={<Bell size={18} color={theme.colors.textMuted} />} label="Notifications" onPress={() => {}} />
      </Section>

      <Section title="More">
        <Row icon={<Heart size={18} color={theme.colors.textMuted} />} label="My Wishlist" onPress={() => router.push('/wishlist')} />
        <Row icon={<Shield size={18} color={theme.colors.textMuted} />} label="Privacy & security" onPress={() => {}} />
        <Row icon={<HelpCircle size={18} color={theme.colors.textMuted} />} label="Help & support" onPress={() => {}} />
      </Section>

      <TouchableOpacity style={styles.signOutBtn} onPress={handleSignOut}>
        <LogOut size={18} color={theme.colors.error} />
        <Text style={styles.signOutText}>Sign out</Text>
      </TouchableOpacity>

      <Text style={styles.version}>NKK Store v1.0.0</Text>
    </ScrollView>
  );
}

function Stat({ label, value, icon }: { label: string; value: string; icon: React.ReactNode }) {
  return (
    <View style={styles.stat}>
      {icon}
      <Text style={styles.statValue}>{value}</Text>
      <Text style={styles.statLabel}>{label}</Text>
    </View>
  );
}

function Section({ title, children }: { title: string; children: React.ReactNode }) {
  return (
    <View style={styles.section}>
      <Text style={styles.sectionTitle}>{title}</Text>
      <View style={styles.sectionBody}>{children}</View>
    </View>
  );
}

function Row({ icon, label, onPress }: { icon: React.ReactNode; label: string; onPress: () => void }) {
  return (
    <TouchableOpacity style={styles.row} activeOpacity={0.7} onPress={onPress}>
      <View style={styles.rowIcon}>{icon}</View>
      <Text style={styles.rowLabel}>{label}</Text>
      <ChevronRight size={18} color={theme.colors.neutral[400]} />
    </TouchableOpacity>
  );
}

const styles = StyleSheet.create({
  container: { flex: 1, backgroundColor: theme.colors.background },
  header: {
    alignItems: 'center',
    paddingTop: 56,
    paddingBottom: 20,
    backgroundColor: theme.colors.primary,
    borderBottomLeftRadius: theme.radius.xl,
    borderBottomRightRadius: theme.radius.xl,
  },
  avatar: {
    width: 80,
    height: 80,
    borderRadius: 40,
    backgroundColor: 'rgba(255,255,255,0.25)',
    alignItems: 'center',
    justifyContent: 'center',
    borderWidth: 2,
    borderColor: 'rgba(255,255,255,0.6)',
  },
  avatarText: { fontSize: 28, fontFamily: theme.fonts.bold, color: '#fff' },
  userName: { fontSize: 16, fontFamily: theme.fonts.bold, color: '#fff', marginTop: 12 },
  userSub: { fontSize: 12, color: 'rgba(255,255,255,0.85)', marginTop: 2, fontFamily: theme.fonts.regular },
  statsRow: {
    flexDirection: 'row',
    backgroundColor: theme.colors.surface,
    borderRadius: theme.radius.lg,
    marginHorizontal: 16,
    marginTop: -16,
    paddingVertical: 16,
    ...theme.shadows.float,
  },
  stat: { flex: 1, alignItems: 'center', gap: 4 },
  statValue: { fontSize: 18, fontFamily: theme.fonts.bold, color: theme.colors.text },
  statLabel: { fontSize: 11, color: theme.colors.textMuted, fontFamily: theme.fonts.medium },
  section: { marginTop: 24, paddingHorizontal: 16 },
  sectionTitle: { fontSize: 13, color: theme.colors.textMuted, fontFamily: theme.fonts.bold, marginBottom: 8, marginLeft: 4 },
  sectionBody: {
    backgroundColor: theme.colors.surface,
    borderRadius: theme.radius.lg,
    overflow: 'hidden',
    ...theme.shadows.card,
  },
  row: {
    flexDirection: 'row',
    alignItems: 'center',
    paddingVertical: 14,
    paddingHorizontal: 16,
    gap: 12,
    borderBottomWidth: 1,
    borderBottomColor: theme.colors.border,
  },
  rowIcon: {
    width: 32,
    height: 32,
    borderRadius: 16,
    backgroundColor: theme.colors.neutral[100],
    alignItems: 'center',
    justifyContent: 'center',
  },
  rowLabel: { flex: 1, fontSize: 14, color: theme.colors.text, fontFamily: theme.fonts.medium },
  signOutBtn: {
    flexDirection: 'row',
    alignItems: 'center',
    justifyContent: 'center',
    gap: 8,
    marginHorizontal: 16,
    marginTop: 24,
    paddingVertical: 14,
    borderRadius: theme.radius.md,
    backgroundColor: '#FEE2E2',
  },
  signOutText: { fontSize: 14, fontFamily: theme.fonts.bold, color: theme.colors.error },
  version: { textAlign: 'center', fontSize: 11, color: theme.colors.textMuted, marginTop: 16, fontFamily: theme.fonts.regular },
});
